            JUN START

INC_RP0,    ISZ R0 NOINC0
            INC R1
NOINC0,     BBL 0

INC_RP1,    LDM 2
            XCH R2
            ADD 2
            XCH R2
            JCN 0b1010 NOINC1
            INC R3
NOINC1,     BBL 0

SAVE_RP1,   LD R3
            SRC R0R1
            WRM
            JMS INC_RP0
            LD R2
            SRC R0R1
            WRM
            JMS INC_RP0
            BBL 0

START,      FIM R0R1 0 ; address pointer
            FIM R2R3 2 ; prime
TEST,       JMS SAVE_RP1
            JMS INC_RP1
            JUN TEST

STOP,       JUN STOP
