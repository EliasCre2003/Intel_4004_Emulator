            JUN START

INC_P1,     LDM 2
            XCH R2
            ADD 2
            XCH R2
            JCN 0b1010 NOCARRY1
            INC R3
NOCARRY1,   BBL 0

INC_P2,     ISZ R4 NOCARRY2
            INC R5
NOCARRY2,   BBL 0

INC_P3,     ISZ R6 NOCARRY3
            INC R5
NOCARRY3,   BBL 0

LOAD_P0,    LD  R1
            SRC P0
            RDM
            JMS INC_P2
            LD  R1
            SRC P0
            RDM
            JMS INC_P2
            BBL 0

SAVE_P1,    LD  R3
            SRC P3
            WRM
            JMS INC_P3
            LD  R2
            SRC P3
            WRM
            JMS INC_P3
            BBL 0

START,      FIM P0 0 ; comp prime
            FIM P1 2 ; prime
            FIM P2 0 ; load address
            FIM P3 0 ; save address
TEST,       JMS SAVE_P1
            JMS INC_P1
            JUN TEST

STOP,       JUN STOP
